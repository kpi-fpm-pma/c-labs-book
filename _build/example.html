#include <stdio.h>

int main(void) 
{
  double a[100];
  int n, i;
  printf("Уведіть кількість чисел n =");
  scanf("%d", &n);
  if(n > (sizeof a) / sizeof(double))
  {
    printf("Забагато елементів\n");
    return 0;
  }
  for(i = 0; i < n; i++)
  {
    printf("a[%d] = ", i);
    scanf("%lf", &a[i]);
  }
  for (i = 0; i < n; ++i)
  {
    printf("%f\n", a[i]);
  }  
}
